The Experiment: Nature Edition

A game made in 4 weeks with 2 developers

Escape the laboratory in VR!

In this immersive VR experience, you find yourself trapped in an abandoned and overgrown laboratory. Explore the environment, repair doors, and escape before the hostile robots catch up to you! With the Meta Quest 3, you’ll feel the suspense and danger around every corner. Can you survive and make it out in time?

                            public class OpenDoor : MonoBehaviour
                            {
                                [SerializeField] Transform door;
                                //[SerializeField] Transform socket;
                                [SerializeField] public GameObject handle;
                                private float openSpeed = 0.5f;
                                private bool handleInPlace = false;
                            
                                private Vector3 startPos;
                                private Vector3 targetPos;
                            
                                public AudioSource Audio;
                                public AudioClip openDoor;
                                public AudioClip WrongDoor;
                            
                                void Start()
                                {
                                    startPos = door.position;
                                    targetPos = new Vector3(startPos.x, startPos.y + 3f, startPos.z);
                                }
                            
                                void Update()
                                {
                                    if (handleInPlace == true)
                                    {
                                        DoorOpening();
                                    }
                                }
                            
                                void DoorOpening()
                                {
                                    door.position = Vector3.Lerp(door.position, targetPos, Time.deltaTime * openSpeed);
                                }
                            
                                private void OnTriggerEnter(Collider other)
                                {
                                    if (other.gameObject == handle)
                                    {
                                        StartCoroutine(WaitBeforeOpening());
                                        handle.transform.SetParent(door);
                            
                                        if(this.gameObject.name == "FirstDoor")
                                        {
                                            WaveSpawner.Instance.Door1Opened = true;
                                        }
                                        if (this.gameObject.name == "SecondDoor")
                                        {
                                            WaveSpawner.Instance.Door2Opened = true;
                                        }
                                        if (this.gameObject.name == "LastDoor")
                                        {
                                            WaveSpawner.Instance.FinalDoorOpened = true;
                                        }
                                    }
                                    if(other.gameObject != handle && other.CompareTag("Gear"))
                                    {
                                        Audio.clip = WrongDoor;
                                        Audio.Play();
                                    }
                                }
                            
                                private IEnumerator WaitBeforeOpening()
                                {
                                    yield return new WaitForSeconds(3f);
                                    handleInPlace = true;
                                    Audio.clip = openDoor;
                                    Audio.Play();
                                }
                            }
                    

                            public class FadeIn : MonoBehaviour
                            {
                                [SerializeField] private GameObject sphere;
                                float fadeAmount = 0;
                            
                                public void FadeBegin()
                                {
                                    {
                                        sphere.SetActive(true);
                                        StartCoroutine(BlackScreenFadeIn());
                                    }
                                }
                            
                                public IEnumerator BlackScreenFadeIn()
                                {
                                    while (fadeAmount < 1)
                                    {
                                        sphere.GetComponent().material.color = new Color(0, 0, 0, fadeAmount);
                                        yield return new WaitForSeconds(0.03f);
                                        fadeAmount += 0.015f;
                            
                                        if (fadeAmount >= 0.95f)
                                        {
                                            SceneManager.LoadScene("GameOver");
                                        }
                                    }
                                }
                        
                    

                                        public class BloodVial : MonoBehaviour
                                        {
                                            private void OnTriggerEnter(Collider other)
                                            {
                                                if(other.CompareTag("Player"))
                                                {
                                                    PlayerHealth PH = other.GetComponent();
                                                    if(PH.Phealth != 100)
                                                    {
                                                        PH.Phealth += 50;
                                                    }

                                                    if(PH.Phealth > 100)
                                                    {
                                                        PH.Phealth = 100;
                                                    }

                                                    Destroy(this.gameObject);
                                                }
                                            }
                                        }
